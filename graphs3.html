get_latest_value = function(input){
  input %>%
    arrange(desc(date)) %>%
    slice(1) %>%
    pull(value)
}

get_latest_date = function(input){
  input %>%
    arrange(desc(date)) %>%
    slice(1) %>%
    pull(date) -> temp
    paste(lubridate::month(temp, label = TRUE),
            lubridate::day(temp),
            lubridate::year(temp))
}

    aes(x = date,
        y = value,

              aes(xmin = start, xmax = end, ymin = -Inf, ymax = Inf),
